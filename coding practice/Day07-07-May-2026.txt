Ansible - Setup Nginx with Custom Config
- name: Setup Nginx with Custom Config
  hosts: all
  become: yes
  tasks:
    - name: Install Nginx
      yum:
        name: nginx
        state: present
    - name: Copy custom nginx.config
      copy:
        src: nginx.config
        dest: /etc/nginx/nginx.config
    - name: Start nginx service
      service:
        name: nginx
        state: started
        enabled: yes
